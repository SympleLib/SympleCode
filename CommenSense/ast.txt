ty ->
	- str base
	- int ptrCount

param ->
	- ty
	- str name
	- expr init

field ->
	- enum vis
	- ty
	- str name
	- expr init

stmt --

struct ->
	- enum vis
	- str name
	- field[] fields

declFunc ->
	- enum vis
	- ty retTy
	- str name
	- ty params

func ->
	- enum vis
	- ty retTy
	- str name
	- param[] params
	- stmt[] body

declVar ->
	- enum vis
	- ty
	- str name

var ->
	- enum vis
	- ty
	- str name
	- expr init

exprStmt ->
	- expr

expr --

binExpr ->
	- expr left
	- enum op
	- expr right

unExpr ->
	- expr operand
	- enum op

funcptr ->
	- str func
	- ty[] params

varExpr ->
	- str name

strLit ->
	- str val

intLit ->
	- int val

fLit ->
	- float val
