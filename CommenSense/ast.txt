module ->
	- str name
	- stmt[] members

ty ->
	- str base
	- int ptrCount

param ->
	- ty
	- str name
	- expr init

field ->
	- enum vis
	- ty
	- str name
	- expr init

stmt --

struct ->
	- enum vis
	- str name
	- field[]

impl ->
	- enum vis
	- str name
	- func[]

class ->
	- enum vis
	- str name
	- field[]
	- func[]

declFunc ->
	- enum vis
	- ty retTy
	- str name
	- ty params

func ->
	- enum vis
	- ty retTy
	- str name
	- param[]
	- stmt[] body

declVar ->
	- enum vis
	- ty
	- str name

var ->
	- enum vis
	- ty
	- str name
	- expr init

exprStmt ->
	- expr

expr --

biExpr ->
	- expr left
	- enum op
	- expr right

unExpr ->
	- expr operand
	- enum op

funcptr ->
	- str func
	- ty[] params

varExpr ->
	- str name

strLit ->
	- str val

intLit ->
	- int val

fLit ->
	- float val
